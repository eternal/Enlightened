00778 // This version doesn't call ctor or dtor.
00779 template< typename TYPE >
00780 HRESULT CGrowableArray<TYPE>::SetSizeInternal( int nNewMaxSize )
00781 {
00782     if( nNewMaxSize < 0 )
00783     {
00784         assert( false );
00785         return E_INVALIDARG;
00786     }
00787 
00788     if( nNewMaxSize == 0 )
00789     {
00790         // Shrink to 0 size & cleanup
00791         if( m_pData )
00792         {
00793             free( m_pData );
00794             m_pData = NULL;
00795         }
00796 
00797         m_nMaxSize = 0;
00798         m_nSize = 0;
00799     }
00800     else if( m_pData == NULL || nNewMaxSize > m_nMaxSize )
00801     {
00802         // Grow array
00803         int nGrowBy = ( m_nMaxSize == 0 ) ? 16 : m_nMaxSize;
00804         nNewMaxSize = __max( nNewMaxSize, m_nMaxSize + nGrowBy );
00805 
00806         TYPE* pDataNew = (TYPE*) realloc( m_pData, nNewMaxSize * sizeof(TYPE) );
00807         if( pDataNew == NULL )
00808             return E_OUTOFMEMORY;
00809 
00810         m_pData = pDataNew;
00811         m_nMaxSize = nNewMaxSize;
00812     }
00813 
00814     return S_OK;
00815 }
